CREATE TABLE "SYSTEM"."JOBSEEKER_DETAILS" 
   (	"JSID" NUMBER(10,0) NOT NULL ENABLE, 
	"JSNAME" VARCHAR2(20 BYTE), 
	"JSADDRS" VARCHAR2(20 BYTE), 
	"JSPPHOTPPATH" VARCHAR2(70 BYTE), 
	"JSRESUMEPATH" VARCHAR2(70 BYTE), 
	 CONSTRAINT "JOBSEEKER_DETAILS_PK" PRIMARY KEY ("JSID"));
	 
	 CREATE SEQUENCE  "SYSTEM"."JSID_SEQ"  MINVALUE 1 MAXVALUE 100000 INCREMENT BY 1 START WITH 21 CACHE 20 NOORDER  NOCYCLE ;
