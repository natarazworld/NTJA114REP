CREATE TABLE "SYSTEM"."CORONA_PATIENT" 
   (	"PATIENT_ID" NUMBER(10,0) NOT NULL ENABLE, 
	"PATIENT_NAME" VARCHAR2(20 BYTE), 
	"PATIENT_ADDRS" VARCHAR2(20 BYTE), 
	"PATIENT_MOBILENO" VARCHAR2(20 BYTE), 
	 CONSTRAINT "CORONA_PATIENT_PK" PRIMARY KEY ("PATIENT_ID"))
	 
	 CREATE SEQUENCE  "SYSTEM"."PATIENT_ID_SEQ"  MINVALUE 1 MAXVALUE 1000 INCREMENT BY 1 START WITH 1 CACHE 20 NOORDER  NOCYCLE ;
