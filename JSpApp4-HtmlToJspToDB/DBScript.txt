### sequence
CREATE SEQUENCE  "SYSTEM"."SNO_SEQ"  MINVALUE 1 MAXVALUE 1000 INCREMENT BY 1 START WITH 4 CACHE 20 NOORDER  NOCYCLE ;

### DB table
CREATE TABLE "SYSTEM"."JSP_STUDENT" 
   (	"SNO" NUMBER(10,0) NOT NULL ENABLE, 
	"SNAME" VARCHAR2(20 BYTE), 
	"SADD" VARCHAR2(20 BYTE), 
	"COURSE" VARCHAR2(20 BYTE), 
	 CONSTRAINT "JSP_STUDENT_PK" PRIMARY KEY ("SNO"));
